import express from "express"; import cors from "cors"; import fetch from "node-fetch"; import { WebSocketServer } from "ws"; const app = express(); app.use(cors()); // ================= SERVER ================= const server = app.listen(3000, () => { console.log("⚡ ARBIMINE SPOT ARBITRAGE ENGINE RUNNING ON PORT 3000"); }); // ================= WEBSOCKET ================= const wss = new WebSocketServer({ server }); // ================= COINS ================= const COINS = [ "BTCUSDT","ETHUSDT","SOLUSDT","BNBUSDT","XRPUSDT", "ADAUSDT","DOGEUSDT","AVAXUSDT","DOTUSDT","LINKUSDT" ]; // ================= EXCHANGE FETCHERS ================= async function binance(symbol) { const r = await fetch(`https://api.binance.com/api/v3/ticker/price?symbol=${symbol}`); const d = await r.json(); return parseFloat(d.price); } async function kucoin(symbol) { const coin = symbol.replace("USDT", ""); const r = await fetch(`https://api.kucoin.com/api/v1/market/orderbook/level1?symbol=${coin}-USDT`); const d = await r.json(); return parseFloat(d.data?.price); } async function bybit(symbol) { const r = await fetch(`https://api.bybit.com/v5/market/tickers?category=spot&symbol=${symbol}`); const d = await r.json(); return parseFloat(d.result?.list?.[0]?.lastPrice); } async function okx(symbol) { const r = await fetch(`https://www.okx.com/api/v5/market/ticker?instId=${symbol}`); const d = await r.json(); return parseFloat(d.data?.[0]?.last); } // ================= SPOT ARBITRAGE ENGINE ================= async function scanSpot() { const results = []; await Promise.all( COINS.map(async (coin) => { const prices = {}; const res = await Promise.allSettled([ binance(coin), kucoin(coin), bybit(coin), okx(coin) ]); const names = ["binance", "kucoin", "bybit", "okx"]; res.forEach((r, i) => { if (r.status === "fulfilled" && r.value > 0) { prices[names[i]] = r.value; } }); const exchanges = Object.keys(prices); if (exchanges.length < 2) return; let buy = exchanges[0]; let sell = exchanges[0]; for (let ex of exchanges) { if (prices[ex] < prices[buy]) buy = ex; if (prices[ex] > prices[sell]) sell = ex; } const buyPrice = prices[buy]; const sellPrice = prices[sell]; let spread = ((sellPrice - buyPrice) / buyPrice) * 100; // REAL SPOT FEES + SLIPPAGE const fee = 0.2; const slippage = 0.1; const netProfit = spread - fee - slippage; if (netProfit > 0.25) { results.push({ coin, buyEx: buy, sellEx: sell, buyPrice, sellPrice, profit: netProfit.toFixed(2) }); } }) ); return results.sort((a, b) => b.profit - a.profit); } // ================= LIVE ENGINE ================= setInterval(async () => { const data = await scanSpot(); wss.clients.forEach(client => { if (client.readyState === 1) { client.send(JSON.stringify(data)); } }); }, 3000); // ================= FRONTEND (IN SAME FILE) ================= app.get("/", (req, res) => { res.send(` ARBIMINE SPOT ENGINE

⚡ LIVE SPOT ARBITRAGE ENGINE

Connecting...
`); });
