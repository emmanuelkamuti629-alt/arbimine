const express = require("express"); const axios = require("axios"); const cors = require("cors"); const app = express(); app.use(cors()); // ============================= // SAFE FETCH ORDER BOOK // ============================= async function fetchBook(name, url, parser) { try { const res = await axios.get(url); const d = parser(res.data); if (!d || !d.bid || !d.ask) return null; return { ex: name, bid: parseFloat(d.bid), ask: parseFloat(d.ask) }; } catch { return null; } } // ============================= // API: REAL ORDER BOOK SCAN // ============================= app.get("/api/scan", async (req, res) => { try { const books = await Promise.all([ fetchBook("Binance", "https://api.binance.com/api/v3/ticker/bookTicker?symbol=BTCUSDT", d => ({ bid: d.bidPrice, ask: d.askPrice }) ), fetchBook("KuCoin", "https://api.kucoin.com/api/v1/market/orderbook/level1?symbol=BTC-USDT", d => ({ bid: d.data.bestBid, ask: d.data.bestAsk }) ), fetchBook("OKX", "https://www.okx.com/api/v5/market/ticker?instId=BTC-USDT", d => ({ bid: d.data[0].bidPx, ask: d.data[0].askPx }) ), fetchBook("Bybit", "https://api.bybit.com/v5/market/tickers?category=spot&symbol=BTCUSDT", d => ({ bid: d.result.list[0].bid1Price, ask: d.result.list[0].ask1Price }) ), fetchBook("Coinbase", "https://api.exchange.coinbase.com/products/BTC-USD/book?level=1", d => ({ bid: d.bids[0][0], ask: d.asks[0][0] }) ), fetchBook("Kraken", "https://api.kraken.com/0/public/Depth?pair=XBTUSD", d => { const pair = Object.keys(d.result)[0]; return { bid: d.result[pair].bids[0][0], ask: d.result[pair].asks[0][0] }; } ) ]); const valid = books.filter(b => b); if (valid.length < 2) { return res.json({ status: "loading market data..." }); } // ============================= // BEST EXECUTABLE BUY/SELL // ============================= const bestBuy = valid.reduce((a, b) => a.ask < b.ask ? a : b); const bestSell = valid.reduce((a, b) => a.bid > b.bid ? a : b); const spread = ((bestSell.bid - bestBuy.ask) / bestBuy.ask) * 100; // fees estimate (realistic average) const fee = 0.2; // 0.2% const net = spread - fee; res.json({ coin: "BTC/USDT", buyExchange: bestBuy.ex, sellExchange: bestSell.ex, buyPrice: bestBuy.ask, sellPrice: bestSell.bid, grossSpread: spread.toFixed(3), netProfit: net.toFixed(3), exchangesActive: valid.length, signal: net > 0.3 ? "🔥 TRADE OPPORTUNITY" : "NO TRADE" }); } catch (e) { res.status(500).json({ error: "scan failed" }); } }); // ============================= // FRONTEND (SIMPLE UI) // ============================= app.get("/", (req, res) => { res.send(` ARB PRO

`); }); // ============================= app.listen(process.env.PORT || 3000, () => console.log("ARBITRAGE ENGINE RUNNING") );
