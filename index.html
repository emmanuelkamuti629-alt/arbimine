const express = require("express"); const axios = require("axios"); const app = express(); const PORT = process.env.PORT || 3000; // ---------------- NORMALIZER ---------------- const norm = (s = "") => s.replace(/[-/]/g, "").toUpperCase(); // ---------------- EXCHANGES ---------------- // BINANCE async function binance() { const r = await axios.get("https://api.binance.com/api/v3/ticker/price"); return r.data.map(c => ({ symbol: c.symbol, price: +c.price, exchange: "Binance" })); } // KUCOIN async function kucoin() { const r = await axios.get("https://api.kucoin.com/api/v1/market/allTickers"); return r.data.data.ticker.map(c => ({ symbol: norm(c.symbol), price: +c.last, exchange: "KuCoin" })); } // OKX async function okx() { const r = await axios.get("https://www.okx.com/api/v5/market/tickers?instType=SPOT"); return r.data.data.map(c => ({ symbol: norm(c.instId), price: +c.last, exchange: "OKX" })); } // BYBIT async function bybit() { const r = await axios.get("https://api.bybit.com/v5/market/tickers?category=spot"); return r.data.result.list.map(c => ({ symbol: norm(c.symbol), price: +c.lastPrice, exchange: "Bybit" })); } // GATE async function gate() { const r = await axios.get("https://api.gateio.ws/api/v4/spot/tickers"); return r.data.map(c => ({ symbol: norm(c.currency_pair), price: +c.last, exchange: "Gate" })); } // MEXC async function mexc() { const r = await axios.get("https://api.mexc.com/api/v3/ticker/price"); return r.data.map(c => ({ symbol: c.symbol, price: +c.price, exchange: "MEXC" })); } // BITGET async function bitget() { const r = await axios.get("https://api.bitget.com/api/spot/v1/market/tickers"); return r.data.data.map(c => ({ symbol: norm(c.symbol), price: +c.last, exchange: "Bitget" })); } // HTX (HUOBI) async function htx() { const r = await axios.get("https://api.huobi.pro/market/tickers"); return r.data.data.map(c => ({ symbol: norm(c.symbol), price: +c.close, exchange: "HTX" })); } // KRAKEN (limited pairs) async function kraken() { const r = await axios.get("https://api.kraken.com/0/public/Ticker?pair=XBTUSD,ETHUSD"); const data = r.data.result; return Object.keys(data).map(k => ({ symbol: norm(k), price: +data[k].c[0], exchange: "Kraken" })); } // COINBASE (limited) async function coinbase() { const r = await axios.get("https://api.exchange.coinbase.com/products/BTC-USD/ticker"); return [{ symbol: "BTCUSD", price: +r.data.price, exchange: "Coinbase" }]; } // ---------------- FETCH ALL ---------------- async function getAll() { const data = await Promise.all([ binance(), kucoin(), okx(), bybit(), gate(), mexc(), bitget(), htx(), kraken(), coinbase() ]); return data.flat(); } // ---------------- ARBITRAGE ENGINE ---------------- function scan(data) { const map = {}; data.forEach(c => { if (!map[c.symbol]) map[c.symbol] = []; map[c.symbol].push(c); }); let results = []; for (let s in map) { const arr = map[s]; if (arr.length < 2) continue; arr.sort((a,b)=>a.price-b.price); const low = arr[0]; const high = arr[arr.length - 1]; const spread = ((high.price - low.price) / low.price) * 100; // realistic filter (fees + noise buffer) if (spread > 0.7) { results.push({ symbol: s, buy: low, sell: high, spread: spread.toFixed(2) }); } } return results.slice(0, 25); } // ---------------- API ---------------- app.get("/scan", async (req, res) => { try { const data = await getAll(); res.json(scan(data)); } catch (e) { res.status(500).json({ error: "scan failed" }); } }); // ---------------- UI (SINGLE FILE FRONTEND) ---------------- app.get("/", (req, res) => { res.send(` Arbitrage Scanner

10-Exchange Crypto Arbitrage Scanner

Loading...
`); }); // ---------------- START ---------------- app.listen(PORT, () => { console.log("Arbitrage Scanner running on http://localhost:" + PORT); });
